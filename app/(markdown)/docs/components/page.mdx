## Usage Patterns

### Basic Import
```tsx
import { AnimateText } from '@/components/animations'
import { Calendar } from '@/components/cards'
```

### Category Import
```tsx
import { AnimateText, Counter, Cursor } from '@/components/animations'
```

### Component Composition
```tsx
export default function Dashboard() {
  return (
    <div className="space-y-6">
      <AnimateText words={["Welcome", "to", "Dashboard"]} />
      <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
        <Calendar />
        <Counter min={0} max={1000} initial={250} />
      </div>
    </div>
  )
}
```

## Customization

### Theme Integration
All components respect your theme configuration:

```tsx
// Components automatically adapt to light/dark mode
<AnimateText 
  words={["Light", "Dark", "Mode"]}
  className="text-foreground" // Uses theme colors
/>
```

### Custom Styling
Components accept custom classes for easy customization:

```tsx
<Counter 
  className="bg-primary text-primary-foreground rounded-lg p-4"
  min={0} 
  max={100} 
/>
```
